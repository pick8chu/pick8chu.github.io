const inputs = `............
........0...
.....0......
.......0....
....0.......
......A.....
............
............
........A...
.........A..
............
............`.split('\n');

const antenna = {};
const antinodes = new Set();

for(let i = 0; i < inputs.length; i++) {
  for(let j = 0; j < inputs[i].length; j++) {
    if(inputs[i][j] !== '.') {
      if(!antenna[inputs[i][j]]) antenna[inputs[i][j]] = [];
      antenna[inputs[i][j]].push([i, j]);

      if(!antinodes[inputs[i][j]]) antinodes[inputs[i][j]] = new Set();
    }
  }
}


for(const key of Object.keys(antenna)) {
  const tempArr = antenna[key];
  for(let i = 0; i < tempArr.length; i++) {
    for(let j = 0; j < tempArr.length; j++) {
      if(i === j) continue;

      diff = [tempArr[i][0] - tempArr[j][0], tempArr[i][1] - tempArr[j][1]];
      // console.log(`from ${tempArr[i]} to ${tempArr[j]} diff ${diff}`);
      
      newI = tempArr[i][0] + diff[0];
      newJ = tempArr[i][1] + diff[1];

      // console.log(`newI, newJ : ${newI}, ${newJ}`);
      if(newI < 0 || newI >= inputs.length || newJ < 0 || newJ >= inputs[0].length) continue;

      // console.log('ðŸ‘†added')
      antinodes.add(`${newI}, ${newJ}`);
    }
  }
}

const inputs = `............
........0...
.....0......
.......0....
....0.......
......A.....
............
............
........A...
.........A..
............
............`.split('\n');

const antenna = {};
const antinodes = new Set();

for(let i = 0; i < inputs.length; i++) {
  for(let j = 0; j < inputs[i].length; j++) {
    if(inputs[i][j] !== '.') {
      if(!antenna[inputs[i][j]]) antenna[inputs[i][j]] = [];
      antenna[inputs[i][j]].push([i, j]);

      if(!antinodes[inputs[i][j]]) antinodes[inputs[i][j]] = new Set();
    }
  }
}


for(const key of Object.keys(antenna)) {
  const tempArr = antenna[key];
  for(let i = 0; i < tempArr.length; i++) {
    for(let j = 0; j < tempArr.length; j++) {
      if(i === j) continue;
      antinodes.add(`${tempArr[i][0]}, ${tempArr[i][1]}`);

      diff = [tempArr[i][0] - tempArr[j][0], tempArr[i][1] - tempArr[j][1]];
      // console.log(`from ${tempArr[i]} to ${tempArr[j]} diff ${diff}`);
      

      newI = tempArr[i][0] + diff[0];
      newJ = tempArr[i][1] + diff[1];

      // console.log(`newI, newJ : ${newI}, ${newJ}`);
      while(!(newI < 0 || newI >= inputs.length || newJ < 0 || newJ >= inputs[0].length)) {
        antinodes.add(`${newI}, ${newJ}`);
        newI = newI + diff[0];
        newJ = newJ + diff[1];
      }

      // console.log('ðŸ‘†added')
    }
  }
}
