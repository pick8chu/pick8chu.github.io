1st part

```js
const inputs = `190: 10 19
3267: 81 40 27
83: 17 5
156: 15 6
7290: 6 8 6 15
161011: 16 10 13
192: 17 8 14
21037: 9 7 18 13
292: 11 6 16 20`.split('\n').map(line => line.split(': '));

let furfilled = new Set();

const dfs = (current, numbers, target,) => {
  if(numbers.length === 0) {
    if(current === target) {
      furfilled.add(current);
    }
    return;
  }

  dfs(current + numbers[0], numbers.slice(1), target);
  dfs(current * numbers[0], numbers.slice(1), target);
}

for(const [index, input] of inputs) {
  const target = Number(index);
  const numbers = input.split(' ').map(Number);
  dfs(0, numbers, target); // <-- this is problem since 0 * anything = 0
}

console.log([...furfilled].reduce((acc, cur) => acc + cur, 0));
```

2nd part

```js
const inputs = `190: 10 19
3267: 81 40 27
83: 17 5
156: 15 6
7290: 6 8 6 15
161011: 16 10 13
192: 17 8 14
21037: 9 7 18 13
292: 11 6 16 20`.split('\n').map(line => line.split(': '));

let furfilled = new Set();

const dfs = (current, numbers, target,) => {
  if(target < current) return;
  if(furfilled.has(target)) return;
  
  if(numbers.length === 0) {
    if(current === target) {
      furfilled.add(target);
    }
    return;
  }

  const newNumbers = numbers.slice(1);
  // console.log(current, numbers[0], Number(String(current) + String(numbers[0])))
  dfs(current + numbers[0], newNumbers, target);
  dfs(current * numbers[0], newNumbers, target);
  dfs(Number(String(current) + String(numbers[0])), newNumbers, target);
}

for(const [index, input] of inputs) {
  const target = Number(index);
  const numbers = input.split(' ').map(Number);
  dfs(numbers[0], numbers.slice(1), target);
}

console.log([...furfilled].reduce((acc, cur) => acc + cur, 0));
```
